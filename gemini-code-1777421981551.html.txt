  { id:'classic', name:'Classic',  cost:0,     color:'#f97316', trail:'rgba(249,115,22,', head:'#fbbf24', perkDesc:'Standard Hero', perk:{} },
  { id:'ice',     name:'Cryo',     cost:500,   color:'#38bdf8', trail:'rgba(56,189,248,', head:'#e0f2fe', perkDesc:'Less Friction (Slides)', perk:{ dragMod: 0.992 } },
  { id:'crimson', name:'Crimson',  cost:1200,  color:'#dc2626', trail:'rgba(220,38,38,',  head:'#fca5a5', perkDesc:'+3 Max Bounces', perk:{ extraBounces: 3 } },
  { id:'gold',    name:'Gold',     cost:2500,  color:'#facc15', trail:'rgba(250,204,21,', head:'#fff',    perkDesc:'2x Coin Yield', perk:{ coinMod: 2 } },
  { id:'shadow',  name:'Shadow',   cost:4000,  color:'#6d28d9', trail:'rgba(109,40,217,', head:'#c4b5fd', perkDesc:'Low Gravity', perk:{ gravMod: 0.75 } },
  { id:'ghost',   name:'Ghost',    cost:7500,  color:'#94a3b8', trail:'rgba(148,163,184,',head:'#f1f5f9', perkDesc:'High Bounce Retention', perk:{ bounceRetain: 1.3 } },
  { id:'neon',    name:'Neon',     cost:15000, color:'#39ff14', trail:'rgba(57,255,20,',  head:'#afff9e', perkDesc:'Rocket Start (+30% Pow)', perk:{ powMod: 1.3 } },
  { id:'galaxy',  name:'Cosmic',   cost:50000, color:'#ff00ff', trail:'rgba(255,0,255,',  head:'#00ffff', perkDesc:'Alien Physics (Low Grav+Drag)', perk:{ gravMod: 0.6, dragMod: 0.995 } },
];

let S = {
  coins: 200, best: 0, runs: 0, totalDist: 0,
  upg: { power:1, aero:1, bounce:0, coinr:1 },
  skin: 'classic', unlocked: ['classic'],
};

try { const sv = localStorage.getItem('ch_infinite'); if(sv) Object.assign(S, JSON.parse(sv)); } catch(e){}
function save() { try { localStorage.setItem('ch_infinite', JSON.stringify(S)); } catch(e){} }

function upgCost(k) {
  const d = UPGRADES_DEF[k], lv = S.upg[k];
  // Calculate cost and round to prevent weird decimals
  return Math.round(d.base * Math.pow(d.mult, lv - (k==='bounce'?0:1)));
}
function buyUpgrade(k) {
  const cost = upgCost(k);
  if (S.coins < cost) return;
  S.coins -= cost; S.upg[k]++; save(); updateUI();
}
function buySkin(id) {
  const sk = SKINS_DEF.find(s=>s.id===id);
  if (!sk) return;
  if (S.unlocked.includes(id)) { S.skin = id; save(); updateUI(); return; }
  if (S.coins < sk.cost) return;
  S.coins -= sk.cost; S.unlocked.push(id); S.skin = id; save(); updateUI();
}

function formatNum(n) {
  if (n >= 1e9) return (n/1e9).toFixed(2) + 'B';
  if (n >= 1e6) return (n/1e6).toFixed(2) + 'M';
  return n.toLocaleString();
}

function updateUI() {
  document.getElementById('coins-disp').textContent = formatNum(S.coins);
  document.getElementById('best-disp').textContent = formatNum(S.best);
  document.getElementById('runs-disp').textContent = formatNum(S.runs);
  document.getElementById('total-disp').textContent = formatNum(S.totalDist);

  for (const k of Object.keys(UPGRADES_DEF)) {
    const lv = S.upg[k];
    const cost = upgCost(k);
    // Visual progress bar loops every 10 levels
    const pct = ((lv % 10) / 10) * 100 || (lv > 0 ? 100 : 0); 
    document.getElementById(`${k}-lv`).textContent = `LV ${lv}`;
    document.getElementById(`${k}-bar`).style.width = (pct === 0 && lv > 0 ? 100 : pct) + '%';
    
    const btn = document.getElementById(`${k}-ubtn`);
    btn.innerHTML = `Upgrade — <span id="${k}-ucost">${formatNum(cost)}</span>`;
    btn.disabled = S.coins < cost;
  }

  // Skins
  const grid = document.getElementById('skins-grid');
  grid.innerHTML = '';
  SKINS_DEF.forEach(sk => {
    const owned = S.unlocked.includes(sk.id);
    const eq = S.skin === sk.id;
    const el = document.createElement('div');
    el.className = `skin-card ${eq?'equipped':''} ${!owned?'locked':''}`;
    el.onclick = () => buySkin(sk.id);
    el.innerHTML = `
      <svg class="skin-prev" viewBox="0 0 48 40" xmlns="http://www.w3.org/2000/svg">
        ${skinSVG(sk, 24, 20)}
      </svg>
      <div class="skin-n">${sk.name}</div>
      <div class="skin-perk">${sk.perkDesc}</div>
      ${owned ? '' : `<div class="skin-c">${formatNum(sk.cost)} Coins</div>`}
    `;
    grid.appendChild(el);
  });
}

function skinSVG(sk, cx, cy) {
  const bc = sk.color, hc = sk.head;
  return `
    <circle cx="${cx}" cy="${cy-12}" r="7" fill="${hc}" opacity="0.95" style="filter: drop-shadow(0 0 5px ${bc})"/>
    <line x1="${cx}" y1="${cy-5}" x2="${cx}" y2="${cy+6}" stroke="${bc}" stroke-width="2.5" stroke-linecap="round"/>
    <line x1="${cx-8}" y1="${cy-1}" x2="${cx+8}" y2="${cy-1}" stroke="${bc}" stroke-width="2" stroke-linecap="round"/>
    <line x1="${cx}" y1="${cy+6}" x2="${cx-6}" y2="${cy+14}" stroke="${bc}" stroke-width="2" stroke-linecap="round"/>
    <line x1="${cx}" y1="${cy+6}" x2="${cx+6}" y2="${cy+14}" stroke="${bc}" stroke-width="2" stroke-linecap="round"/>
  `;
}

const canvas = document.getElementById('gc');
const ctx = canvas.getContext('2d');
const wrap = document.getElementById('game-wrap');

let W, H;
function resize() {
  W = canvas.width  = wrap.clientWidth;
  H = canvas.height = wrap.clientHeight;
}
resize();
window.addEventListener('resize', () => { resize(); if (!gameActive) drawMenu(); });

// Game state
let gameActive = false, aiming = false, launched = false;
let px, py, vx, vy;
let chargeAmt;
let camX, shake = 0;
let dist, bestThisRun, currentCoinsEarned = 0;
let bounces;
let trailPoints = [], clouds = [], groundDeco = [], mountains = [], particles = [];
let animId;

const GROUND_Y_RATIO = 0.75;
function groundY() { return H * GROUND_Y_RATIO; }
const CANNON_X = 120;
function cannonY() { return groundY(); }

function genScenery() {
  clouds = []; groundDeco = []; mountains = [];
  for (let i = 0; i < 15; i++) {
    clouds.push({ x: Math.random() * 8000, y: Math.random() * H * 0.4, w: 80 + Math.random() * 200, h: 30 + Math.random() * 50, speed: 0.05 + Math.random() * 0.15, alpha: 0.2 + Math.random() * 0.4 });
  }
  for(let i=0; i<30; i++) {
    mountains.push({ x: Math.random() * 10000, w: 300 + Math.random()*500, h: 100 + Math.random()*250, color: `hsl(210, 30%, ${15 + Math.random()*10}%)` });
  }
  for (let i = 0; i < 300; i++) {
    groundDeco.push({ x: Math.random() * 15000, type: Math.random() < 0.7 ? 'grass' : 'rock', size: 4 + Math.random() * 15 });
  }
}
genScenery();

// ── DRAW MENU SCREEN ──
function drawMenu() {
  resize();
  ctx.clearRect(0,0,W,H);
  drawSky();
  drawMountains(0);
  drawClouds(0);
  drawGroundDeco(0);
  drawGround(0);
  drawCannon(CANNON_X, cannonY(), Math.PI/5, 0);
  drawStickman(CANNON_X + 60, groundY(), 0);
}

// ── DRAWING PRIMITIVES ──
function drawSky() {
  const g = ctx.createLinearGradient(0,0,0,H);
  g.addColorStop(0, '#050a14'); g.addColorStop(0.4, '#122442'); g.addColorStop(1, '#244534');
  ctx.fillStyle = g; ctx.fillRect(0,0,W,H);
  // Stars
  ctx.fillStyle = '#fff';
  for(let i=0; i<50; i++) {
     let sx = (i * 137) % W, sy = (i * 93) % (H*0.5);
     ctx.globalAlpha = Math.random() * 0.8;
     ctx.fillRect(sx, sy, 1.5, 1.5);
  }
  ctx.globalAlpha = 1;
}

function drawMountains(camX) {
  mountains.forEach(m => {
    const mx = ((m.x - camX * 0.1) % (W + 1000) + W + 1000) % (W + 1000) - 500;
    ctx.fillStyle = m.color;
    ctx.beginPath();
    ctx.moveTo(mx, groundY());
    ctx.lineTo(mx + m.w/2, groundY() - m.h);
    ctx.lineTo(mx + m.w, groundY());
    ctx.fill();
  });
}

function drawClouds(camX) {
  clouds.forEach(c => {
    const cx = ((c.x - camX * 0.25) % (W + 400) + W + 400) % (W + 400) - 200;
    ctx.save();
    ctx.globalAlpha = c.alpha; ctx.fillStyle = '#c8dff5';
    ctx.beginPath(); ctx.ellipse(cx, c.y, c.w*0.5, c.h*0.45, 0, 0, Math.PI*2); ctx.fill();
    ctx.beginPath(); ctx.ellipse(cx-c.w*0.28, c.y+c.h*0.1, c.w*0.32, c.h*0.35, 0, 0, Math.PI*2); ctx.fill();
    ctx.beginPath(); ctx.ellipse(cx+c.w*0.28, c.y+c.h*0.1, c.w*0.3,  c.h*0.32, 0, 0, Math.PI*2); ctx.fill();
    ctx.restore();
  });
}

function drawGround(camX) {
  const gy = groundY();
  const gg = ctx.createLinearGradient(0, gy, 0, H);
  gg.addColorStop(0, '#3a6b28'); gg.addColorStop(0.1, '#1b3812'); gg.addColorStop(1, '#0a1205');
  ctx.fillStyle = gg; ctx.fillRect(0, gy, W, H - gy);
  ctx.fillStyle = '#5a9a3a'; ctx.fillRect(0, gy, W, 6);
  ctx.fillStyle = '#7abd52'; ctx.fillRect(0, gy, W, 2);

  ctx.save();
  ctx.font = '800 12px "Bebas Neue", sans-serif';
  ctx.textAlign = 'center';
  for (let m = 0; m < 50000; m += 100) {
    const sx = m * 4 - camX;
    if (sx < -50 || sx > W + 50) continue;
    ctx.fillStyle = 'rgba(255,255,255,0.15)';
    ctx.fillRect(sx - 1, gy + 8, 2, 12);
    ctx.fillStyle = 'rgba(255,255,255,0.3)';
    ctx.fillText(m + 'M', sx, gy + 35);
  }
  ctx.restore();
}

function drawGroundDeco(camX) {
  groundDeco.forEach(d => {
    const sx = d.x - camX;
    if (sx < -30 || sx > W + 30) return;
    const gy = groundY();
    if (d.type === 'grass') {
      ctx.strokeStyle = '#4a8a30'; ctx.lineWidth = 2; ctx.lineCap = 'round';
      for (let i = -1; i <= 1; i++) {
        ctx.beginPath(); ctx.moveTo(sx + i*5, gy); ctx.lineTo(sx + i*4, gy - d.size * (0.5 + Math.abs(i)*0.4)); ctx.stroke();
      }
    } else {
      ctx.fillStyle = '#222'; ctx.beginPath(); ctx.ellipse(sx, gy + 2, d.size, d.size*0.4, 0, 0, Math.PI*2); ctx.fill();
    }
  });
}

function drawCannon(wx, wy, angle, camX) {
  const sx = wx - camX, sy = wy;
  ctx.save(); ctx.translate(sx, sy - 12);
  
  // Back wheel
  ctx.fillStyle = '#111'; ctx.beginPath(); ctx.arc(-10, 12, 18, 0, Math.PI*2); ctx.fill();

  // Barrel
  ctx.rotate(-angle);
  const barG = ctx.createLinearGradient(-8, -45, 8, -45);
  barG.addColorStop(0, '#444'); barG.addColorStop(0.5, '#777'); barG.addColorStop(1, '#222');
  ctx.fillStyle = barG;
  ctx.beginPath(); ctx.roundRect(-10, -50, 20, 48, 5); ctx.fill();
  ctx.fillStyle = '#111'; ctx.beginPath(); ctx.ellipse(0, -50, 12, 6, 0, 0, Math.PI*2); ctx.fill();
  ctx.fillStyle = '#ff8800'; ctx.beginPath(); ctx.ellipse(0, -50, 8, 4, 0, 0, Math.PI*2); ctx.fill(); // inner glow
  
  // Front wheel/mount
  ctx.rotate(angle);
  ctx.fillStyle = '#1a1a1a'; ctx.beginPath(); ctx.arc(0, 12, 22, 0, Math.PI*2); ctx.fill();
  ctx.strokeStyle = '#333'; ctx.lineWidth = 4; ctx.stroke();
  ctx.fillStyle = '#f97316'; ctx.beginPath(); ctx.arc(0, 12, 5, 0, Math.PI*2); ctx.fill(); // hub
  
  ctx.restore();
}

function drawStickman(wx, wy, camX) {
  const sx = wx - camX, sy = wy;
  const sk = SKINS_DEF.find(s=>s.id===S.skin) || SKINS_DEF[0];
  ctx.save();
  ctx.shadowColor = sk.color; ctx.shadowBlur = 15;
  ctx.fillStyle = sk.head; ctx.beginPath(); ctx.arc(sx, sy - 28, 10, 0, Math.PI*2); ctx.fill();
  ctx.strokeStyle = sk.color; ctx.lineWidth = 3; ctx.lineCap = 'round'; ctx.stroke();
  ctx.beginPath(); ctx.moveTo(sx, sy-18); ctx.lineTo(sx, sy-6); ctx.stroke(); // Body
  ctx.beginPath(); ctx.moveTo(sx-12, sy-15); ctx.lineTo(sx+12, sy-15); ctx.stroke(); // Arms
  ctx.beginPath(); ctx.moveTo(sx, sy-6); ctx.lineTo(sx-8, sy+6); ctx.stroke(); // Legs
  ctx.beginPath(); ctx.moveTo(sx, sy-6); ctx.lineTo(sx+8, sy+6); ctx.stroke();
  ctx.shadowBlur = 0; ctx.restore();
}

function drawTrail(camX) {
  if (trailPoints.length < 2) return;
  const sk = SKINS_DEF.find(s=>s.id===S.skin)||SKINS_DEF[0];
  ctx.save();
  ctx.globalCompositeOperation = 'lighter';
  for (let i = 1; i < trailPoints.length; i++) {
    const t = i / trailPoints.length;
    ctx.fillStyle = sk.trail + (t * 0.6) + ')';
    ctx.beginPath(); ctx.arc(trailPoints[i].x - camX, trailPoints[i].y, t * 8, 0, Math.PI*2); ctx.fill();
  }
  ctx.restore();
}

function drawParticles(camX) {
  ctx.save();
  ctx.globalCompositeOperation = 'lighter';
  particles.forEach(p => {
    ctx.globalAlpha = Math.max(0, p.life);
    ctx.fillStyle = p.color;
    ctx.beginPath(); ctx.arc(p.x - camX, p.y, p.r * p.life, 0, Math.PI*2); ctx.fill();
  });
  ctx.restore();
}

function spawnExplosion(x, y, color, amount, power) {
  for (let i = 0; i < amount; i++) {
    const ang = Math.random() * Math.PI * 2;
    const spd = Math.random() * power;
    particles.push({
      x: x, y: y, vx: Math.cos(ang)*spd, vy: Math.sin(ang)*spd,
      r: 3 + Math.random()*8, life: 1, color: color
    });
  }
}

function startGame() {
  document.getElementById('panel').classList.add('hidden');
  document.getElementById('result-overlay').classList.remove('show');
  document.getElementById('hud').classList.add('show');
  document.getElementById('power-wrap').style.display = 'flex';

  resize();
  gameActive = true; aiming = true; launched = false; bounces = 0;
  trailPoints = []; particles = []; camX = 0; dist = 0; currentCoinsEarned = 0;
  bestThisRun = false; chargeAmt = 0; shake = 0;

  px = CANNON_X; py = cannonY() - 45; vx = 0; vy = 0;
  document.getElementById('hud-dist').textContent = '0m';
  document.getElementById('hud-coins-run').textContent = '0';

  let charging = false, chargePct = 0, chargeDir = 1, autoCharge = true;
  S.runs++; document.getElementById('runs-disp').textContent = formatNum(S.runs);
  if (animId) cancelAnimationFrame(animId);

  const sk = SKINS_DEF.find(s=>s.id===S.skin)||SKINS_DEF[0];
  const perk = sk.perk || {};

  function onPress() { if (!aiming || launched) return; charging = true; autoCharge = false; }
  function onRelease() { if (!aiming || !charging || launched) return; charging = false; fire(chargePct); }

  canvas.addEventListener('mousedown', onPress); canvas.addEventListener('mouseup', onRelease);
  canvas.addEventListener('touchstart', onPress, {passive:true}); canvas.addEventListener('touchend', onRelease, {passive:true});

  function cleanup() {
    canvas.removeEventListener('mousedown', onPress); canvas.removeEventListener('mouseup', onRelease);
    canvas.removeEventListener('touchstart', onPress); canvas.removeEventListener('touchend', onRelease);
  }

  function fire(pct) {
    if (launched) return;
    launched = true; aiming = false;
    document.getElementById('power-wrap').style.display = 'none';

    // Apply scaling and perks
    const basePow  = 15 + S.upg.power * 2.5; 
    const powMod   = perk.powMod || 1;
    const totalPow = (basePow * powMod) * (0.3 + pct * 0.7);
    const angle    = Math.PI / 4.5;
    
    vx = Math.cos(angle) * totalPow;
    vy = -Math.sin(angle) * totalPow;
    shake = 15; // Screen shake!

    const mx = CANNON_X + Math.cos(angle) * 50;
    const my = (cannonY() - 12) - Math.sin(angle) * 50;
    spawnExplosion(mx, my, '#f97316', 40, totalPow * 0.5);
    spawnExplosion(mx, my, '#fff', 20, totalPow * 0.8);
  }

  function loop() {
    animId = requestAnimationFrame(loop);
    
    ctx.save();
    if (shake > 0) {
      ctx.translate((Math.random()-0.5)*shake, (Math.random()-0.5)*shake);
      shake *= 0.85; if(shake < 0.5) shake = 0;
    }
    ctx.clearRect(-100,-100,W+200,H+200);

    if (aiming && !launched) {
      if (autoCharge || charging) {
        chargePct += 0.018 * chargeDir;
        if (chargePct >= 1) { chargePct = 1; chargeDir = -1; }
        if (chargePct <= 0) { chargePct = 0; chargeDir = 1; }
      }
      updatePowerBar(chargePct);
    }

    if (launched) {
      // Infinite aerodynamics scaling formula (approaches 1.0 asymptotically)
      const aeroMod = 1 + S.upg.aero * 0.15;
      const dragBase = 1 - (0.02 / aeroMod);
      const drag = dragBase * (perk.dragMod || 1); // some skins slip through air
      const gravity = 0.42 * (perk.gravMod || 1);

      vx *= drag; vy += gravity;
      px += vx; py += vy;

      const targetCam = px - W * 0.35;
      camX += (targetCam - camX) * 0.15;
      if (camX < 0) camX = 0;

      // Ground collision
      if (py >= cannonY()) {
        py = cannonY();
        const maxBounces = S.upg.bounce + (perk.extraBounces || 0);
        // Bounce retain grows slowly, soft cap around 0.85 to prevent infinite bouncing
        const baseRetain = 0.35 + Math.min(0.5, S.upg.bounce * 0.03); 
        const bounceFactor = baseRetain * (perk.bounceRetain || 1);

        if (Math.abs(vy) > 1.5 && bounces < maxBounces) {
          vy = -Math.abs(vy) * bounceFactor;
          vx *= 0.8;
          bounces++; shake = 8;
          spawnExplosion(px, py, sk.color, 15, Math.abs(vy)*0.5);
        } else {
          // Dead stop
          vx = 0; vy = 0; launched = false;
          const meters = Math.round((px - CANNON_X) / 4);
          endRun(meters); cleanup();
          ctx.restore(); return;
        }
      }

      trailPoints.push({x:px, y:py});
      if (trailPoints.length > 80) trailPoints.shift();

      dist = Math.round((px - CANNON_X) / 4);
      document.getElementById('hud-dist').textContent = dist + 'm';
      
      // Calculate coins dynamically
      const coinMod = perk.coinMod || 1;
      // Linear scaling with upgrade
      const newCoins = Math.floor(dist * 0.5 * (1 + S.upg.coinr * 0.5) * coinMod);
      if (newCoins > currentCoinsEarned) {
        currentCoinsEarned = newCoins;
        document.getElementById('hud-coins-run').textContent = formatNum(currentCoinsEarned);
      }
    }

    particles.forEach(p => { p.x += p.vx; p.y += p.vy; p.vy += 0.2; p.life -= 0.03; p.r *= 0.95; });
    particles = particles.filter(p => p.life > 0);

    drawSky(); drawMountains(camX); drawClouds(camX);
    drawGroundDeco(camX); drawGround(camX);
    drawTrail(camX); drawParticles(camX);

    const cannonAngle = launched ? Math.PI/4.5 : (Math.PI/6 + chargePct * Math.PI/10);
    drawCannon(CANNON_X, cannonY(), cannonAngle, camX);

    if (!launched && aiming) {
      const ang = cannonAngle, barrelLen = 45;
      const spx = CANNON_X + Math.cos(ang) * barrelLen - camX;
      const spy = (cannonY() - 12) - Math.sin(ang) * barrelLen;
      ctx.save(); ctx.shadowColor = sk.color; ctx.shadowBlur = 15; ctx.fillStyle = sk.head;
      ctx.beginPath(); ctx.arc(spx, spy, 10, 0, Math.PI*2); ctx.fill(); ctx.restore();
    } else {
      drawStickman(px, py, camX);
    }
    
    ctx.restore(); // remove shake
  }

  loop();
}

function updatePowerBar(pct) {
  const fill = document.getElementById('power-fill');
  fill.style.width = (pct * 100) + '%';
  const hue = 120 - pct * 120;
  fill.style.background = `linear-gradient(90deg, hsl(${hue+20},90%,55%), hsl(${hue},90%,45%), #fff)`;
  fill.style.boxShadow = `0 0 15px hsl(${hue},90%,50%)`;
}

function endRun(meters) {
  S.coins += currentCoinsEarned;
  S.totalDist += meters;
  const isNewBest = meters > S.best;
  if (isNewBest) S.best = meters;

  save(); updateUI();

  document.getElementById('ro-dist').textContent = formatNum(meters);
  document.getElementById('ro-coins').textContent = `+ ${formatNum(currentCoinsEarned)} COINS`;
  document.getElementById('ro-best-badge').style.display = isNewBest ? 'block' : 'none';
  document.getElementById('result-overlay').classList.add('show');
  document.getElementById('hud').classList.remove('show');
}

function backToMenu() {
  if (animId) cancelAnimationFrame(animId);
  gameActive = false; launched = false; aiming = false;
  document.getElementById('panel').classList.remove('hidden');
  document.getElementById('result-overlay').classList.remove('show');
  document.getElementById('hud').classList.remove('show');
  document.getElementById('power-wrap').style.display = 'none';
  updateUI(); setTimeout(drawMenu, 100);
}
